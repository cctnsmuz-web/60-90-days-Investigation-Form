/** * CCTNS Data Entry Form - Backend Script * ---------------------------------------- * Yeh script Google Sheet ke "Extensions > Apps Script" mein lagana hai. * Iska kaam: web form se aaya hua data validate karna aur Sheet mein * naya row add karna - taaki thana wale log seedha Sheet edit na karein, * sirf form bharein aur yahi script safely Sheet update kare. * * SETUP STEPS (ek baar karna hai): * 1. Apni Google Sheet kholiye * 2. Top menu mein "Extensions" -> "Apps Script" par click karein * 3. Jo default code khula ho usse sab delete karke yeh poora code paste kar dein * 4. Upar "Save" (floppy disk icon) dabayen * 5. "Deploy" -> "New deployment" par click karein * 6. Gear icon (Settings) -> "Web app" select karein * 7. "Who has access" mein "Anyone" select karein (zaroori hai, warna form * kaam nahi karega) * 8. "Deploy" dabayen, Google permission maangega - "Authorize access" * karein, apna Google account select karein, "Advanced" -> "Go to * project (unsafe)" -> "Allow" (yeh normal hai, yeh aapki apni Sheet hai) * 9. Jo "Web app URL" milega (https://script.google.com/macros/s/.../exec) * - usi ko HTML form file mein APPS_SCRIPT_URL wali jagah daalna hai */ const SHEET_NAME = "Sheet1"; // Agar aapki sheet ka tab ka naam alag hai to yahan badal dein function doPost(e) { try { const params = e.parameter; // ---------- Server-side validation (zaroori, kyunki client-side // validation ko koi bypass kar sakta hai) ---------- const required = ["ps_name", "io_name", "fir_no", "fir_date", "inv_type", "mobile"]; for (const field of required) { if (!params[field] || params[field].toString().trim() === "") { return respond(false, `Field '${field}' khaali nahi ho sakta.`); } } const mobile = params.mobile.toString().trim(); if (!/^[6-9]\d{9}$/.test(mobile)) { return respond(false, "Mobile number 10 digit ka hona chahiye aur 6-9 se shuru hona chahiye."); } const invType = params.inv_type.toString().trim(); if (invType !== "60" && invType !== "90") { return respond(false, "Inv_Type sirf 60 ya 90 ho sakta hai."); } const firDate = params.fir_date.toString().trim(); if (!/^\d{2}\/\d{2}\/\d{4}$/.test(firDate)) { return respond(false, "FIR_Date format DD/MM/YYYY mein hona chahiye."); } // ---------- Sheet mein row add karo ---------- const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); if (!sheet) { return respond(false, `Sheet tab '${SHEET_NAME}' nahi mila.`); } sheet.appendRow([ params.ps_name.toString().trim(), params.io_name.toString().trim(), params.fir_no.toString().trim(), firDate, invType, mobile, "" // Chargesheet_Filed - hamesha khaali shuru mein ]); return respond(true, "Data safaltapoorvak save ho gaya."); } catch (err) { return respond(false, "Server error: " + err.toString()); } } function respond(success, message) { return ContentService .createTextOutput(JSON.stringify({ success: success, message: message })) .setMimeType(ContentService.MimeType.JSON); } // Browser mein seedha URL khol kar test karne ke liye (optional, GET request) function doGet(e) { return ContentService .createTextOutput("CCTNS Data Entry backend chal raha hai. Form se POST request bhejen.") .setMimeType(ContentService.MimeType.TEXT); }
